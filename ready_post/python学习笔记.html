定义函数

import math

def quadratic(a,b,c):
    x=(-b+math.sqrt(b*b-4*a*c))/(2*a)
    y=(-b-math.sqrt(b*b-4*a*c))/(2*a)
    return x,y
print(quadratic(2, 3, 1)) # => (-0.5, -1.0)
print(quadratic(1, 3, -4)) # => (1.0, -4.0)
